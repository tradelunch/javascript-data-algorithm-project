# 5. 53. Maximum Subarray



{% embed url="https://leetcode.com/problems/maximum-subarray/description/" %}

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function(nums) {
    
    let maxSum = -Infinity;
    let sum = -Infinity;
    for (let i = 0; i < nums.length; i++) {
        const curr = nums[i];
        sum = Math.max(curr, sum + curr);
        maxSum = Math.max(maxSum, sum);
    }

    return maxSum;
};
```
