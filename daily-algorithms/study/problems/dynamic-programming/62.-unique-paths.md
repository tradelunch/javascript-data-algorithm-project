### 1st try

> time: O(m x n)

> space: O(m x n)

```jsx
/**
 * @param {number} m
 * @param {number} n
 * @return {number}
 */
var uniquePaths = function(m, n) {
    const matrix = new Array(m);

    matrix[0] = new Array(n).fill(1);
    for (let i = 1; i < m; i++) {
        matrix[i] = new Array(n);
        matrix[i][0] = 1;
    }

    for (let i = 1; i < m; i++) {
        for (let j = 1; j < n; j++) {
            matrix[i][j] = matrix[i - 1][j] + matrix[i][j - 1];
        }
    }

    return matrix[m - 1][n - 1];
};
```
