> time: O(N)

> space: O(32) > O(1)

```jsx
/**
 * @param {number} n - a positive integer
 * @return {number} - a positive integer
 */
var reverseBits = function(n) {
    const bi = n.toString(2);
    const paded = bi.padStart(32, '0');
    const reversed = [...paded].reverse().join('');

    return parseInt(reversed, 2);
};
```
